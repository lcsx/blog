# 直方图

import pandas as pd
import numpy as np

df = pd.read_csv('./data/data/HR.csv')

import seaborn as sns
import matplotlib.pyplot as plt



# df = df.where(df["salary"] != 'nme').dropna()
df = df[df["last_evaluation"]<=1][df["salary"]!="nme"][df["department"]!="sale"]

sns.set_style(style = "darkgrid")

sns.set_context(context="poster",font_scale=0.8)


f = plt.figure()
f.add_subplot(1,3,1)   # 添加子图 1行3列第一个

# 绘制displot
# sns.displot(df["satisfaction_level"],bins=10)  #直方图
sns.displot(df["satisfaction_level"],bins=10,kde=True)   #有分布图和直方图


f.add_subplot(1,3,2)   # 添加子图 1行3列第2个
sns.displot(df["last_evaluation"],bins=10)


f.add_subplot(1,3,3)   # 添加子图 1行3列第3个
sns.displot(df["average_monthly_hours"],bins=10)

plt.show()

# 饼图

seaborn中没有饼图，只能用matplotlib绘图

import pandas as pd
import numpy as np

df = pd.read_csv('./data/data/HR.csv')

import seaborn as sns
import matplotlib.pyplot as plt

df = df[df["last_evaluation"]<=1][df["salary"]!="nme"][df["department"]!="sale"]


lbs = df["department"].value_counts().index
# plt.pie(df["department"].value_counts(normalize=True),labels=lbs,autopct="%1.1f%%")


# plt.show()


# 突出一个项
explodes = [0.1 if i== "sales" else 0 for i in lbs]
plt.pie(df["department"].value_counts(normalize=True),labels=lbs,autopct="%1.1f%%",explode = explodes)
plt.show()
