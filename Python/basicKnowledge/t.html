# t分布

t-分布（t-distribution）用于根据小样本来估计呈正态分布且方差未知的总体的均值。如果总体方差已知（例如在样本数量足够多时），则应该用正态分布来估计总体均值。

# 独立t分布检验
# 检验两组数据的均值差异水平

import numpy as np
import scipy.stats as ss


xx = ss.ttest_ind(ss.norm.rvs(size=10),ss.norm.rvs(size=20))
print(xx)
# Ttest_indResult(statistic=0.48605562136576796, pvalue=0.6307101254445986)
# pvalue > 0.05  （以0.05作为显著性水平来比较的话，是可以接受假设的，即两组数的均值是没有差别的）

# 交叉分析
# 题目：分析各部门的离职率【left】之间是否有明显差异
# 使用的方法：独立t检验
# 思路，得到各部门的离职分布，然后两两间求t检验量，并求出p值


import pandas as pd
import numpy as np
import scipy.stats as ss
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('./data/data/HR.csv')
# 去掉异常值数据
df = df[df["last_evaluation"]<=1][df["salary"]!="nme"][df["department"]!="sale"]
# print(df)

# 为了得到各部门的离职率分布，得先groupby一下
# indices属性，获取索引
dp_indices = df.groupby(by="department").indices
# print(dp_indices)
#'sales': array([    0,     1,     2, ..., 14969, 14970, 14971], dtype=int64), 'support': array([   46,    47,    48, ..., 14997, 14998, 14999], dtype=int64), 'technical': array([   35,    36,    37, ..., 14987, 14988, 14989], dtype=int64)} 
# 解读：sales的项在dp中是第14969, 14970, 14971项

# 1.1
# 单个方式
# 根据索引获取值
# sales_values = df["left"].iloc[dp_indices["sales"]].values
# print(sales_values)

# technical_values = df["left"].iloc[dp_indices["technical"]].values
# print(technical_values)
# print(ss.ttest_ind(sales_values,technical_values)[1])

# 1.2
# 推荐方式
# 两两比较
dp_keys = list(dp_indices.keys())
# print(dp_keys)
# dict_keys(['IT', 'RandD', 'accounting', 'hr', 'management', 'marketing', 'product_mng', 'sales', 'support', 'technical'])



# 2.1
# 弄个二维图来展示两两关系
# 声明一个二维矩阵
dp_t_mat = np.zeros([len(dp_keys),len(dp_keys)])
# print(dp_t_mat)
# for i in range(len(dp_keys)):
#     for j in range(len(dp_keys)):
#         p_value = ss.ttest_ind(df["left"].iloc[dp_indices[dp_keys[i]]].values,\
#             df["left"].iloc[dp_indices[dp_keys[j]]].values)[1]
#         dp_t_mat[i][j] = p_value

# 2.2
for i in range(len(dp_keys)):
    for j in range(len(dp_keys)):
        p_value = ss.ttest_ind(df["left"].iloc[dp_indices[dp_keys[i]]].values,\
            df["left"].iloc[dp_indices[dp_keys[j]]].values)[1]
        if p_value <0.05:
            dp_t_mat[i][j] = -1
        else:
            dp_t_mat[i][j] = p_value

# print(dp_t_mat)
sns.heatmap(dp_t_mat,xticklabels=dp_keys,yticklabels=dp_keys)
plt.show()

# 交叉分析.透视表

import pandas as pd
import numpy as np
import scipy.stats as ss
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('./data/data/HR.csv')
# 去掉异常值数据
df = df[df["last_evaluation"]<=1][df["salary"]!="nme"][df["department"]!="sale"]
# print(df)


# 透视表
piv_tb = pd.pivot_table(df,values="left",index=["promotion_last_5years","salary"],\
    columns=["Work_accident"],aggfunc=np.mean)

# print(piv_tb)

#                       medium  0.028986  0.023256

# 解读
# 0.331835  没有工作事故，低薪，离职最多
