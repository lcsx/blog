# 多个不同文件，单个文件多个表

# 在一个大文件集合中查找一组项目

每个公司都在各自不同的业务过程中积累了大量文件，在供应商、客户、内部运营和其他 业务过程中，可能存在着大量历史文件。正如本书已经讨论过的，这些数据可能被保存在 像 CSV 文件这样的带分隔符的平面文件中，可能被保存在 Excel 工作簿和电子表格中，也 可能被保存在其他存储系统中。保存这些文件是有价值的，因为它们可以提供数据用来分 析，有助于跟踪随着时间发生的变化，还可以为决策提供支持。

但是当你有大量历史数据的时候，要找到真正需要的数据是非常困难的。想象一下，假如 你有 300 个 Excel 工作簿和 200 个 CSV 文件（你一直交替使用这两种类型的文件）

当然，你可以打开每个文件，找出你需要的记录，然后将记录复制粘贴到一个新文件中。 但是，这个过程太痛苦了，既浪费时间，又容易出错。这正是展示你刚学会的 Python 编程 技能的一个绝好机会，你可以自动化地完成整个过程，既节省时间，又不会出错。

# 递归遍历所有文件

目录，子目录，.csv/.xlsx/.xls格式

import pandas as pd
import glob
import os,time
from xlrd import open_workbook,xldate_as_tuple

input_path = r"C:\Users\46957\Desktop\pytest\basisofpython\excels"
output_file = r"C:\Users\46957\Desktop\pytest\basisofpython\result\excels\gg1.xlsx"



def walk_dir2(dirname):
    for d in os.listdir(dirname):
        path = os.path.join(dirname, d)
        if os.path.isdir(path):
            for f in walk_dir2(path):
                yield f
        else:
            yield path


filepath = os.path.abspath(input_path)
files = walk_dir2(filepath)
total_score = 0
total_student_number = 0
average_score = 0


for file in files:
    # 如果是csv格式
    if file.split('.')[-1] == 'csv':
        csv_file = pd.read_csv(file, index_col=None,encoding = 'gb2312')
        total_score += csv_file['成绩'].sum()
        total_student_number += len(csv_file['成绩'])
    elif file.split('.')[-1] == 'xlsx' or file.split('.')[-1] == 'xls':
        data_frames = []
        all_worksheets = pd.read_excel(file, sheet_name=None, index_col=None)
        for worksheet_name, data in all_worksheets.items():
            # 筛选列
            # print(data[['姓名','成绩']],worksheet_name)
            data_frames.append(pd.DataFrame(data[['姓名','成绩']], \
            columns=['姓名', '成绩']))
        data_frames = pd.concat(data_frames, axis=0, ignore_index=True)

        total_score += data_frames['成绩'].sum()
        total_student_number += len(data_frames['成绩'])


data = {
    "总分":total_score,
    "学生人数":total_student_number,
    "平均分":float(total_score/total_student_number)
}

fm = pd.DataFrame(data, \
columns=['总分', '学生人数', '平均分'],index=[0])


with pd.ExcelWriter(output_file,mode='a+',engine='openpyxl') as writer:
# with pd.ExcelWriter(output_file,mode='a',engine='openpyxl') as writer:
# writer = pd.ExcelWriter(output_file,mode='a+',engine='openpyxl')
# writer = pd.ExcelWriter(output_file,mode='a',engine='openpyxl')
    fm.to_excel(writer, sheet_name='lcs', \
    index=False)
    writer.save()
