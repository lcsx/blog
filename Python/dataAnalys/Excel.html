# sheet1.cell_type(x,y)

ctype : 0 empty,1 string, 2 number, 3 date, 4 boolean, 5 error。即date的ctype=3

from datetime import date
from xlrd import open_workbook, xldate_as_tuple
from xlwt import Workbook
input_file = r"C:\Users\46957\Desktop\pytest\basisofpython\excels\cj1.xlsx"
output_file = r"C:\Users\46957\Desktop\pytest\basisofpython\excels\cj2.xlsx"
output_workbook = Workbook()
output_worksheet = output_workbook.add_sheet('jan_2013_output')
with open_workbook(input_file) as workbook:
    worksheet = workbook.sheet_by_name('lcsSheet1')
    for row_index in range(worksheet.nrows):
        row_list_output = []
        for col_index in range(worksheet.ncols):
            print(worksheet.cell_type(row_index, col_index))
            if worksheet.cell_type(row_index, col_index) == 3:
                date_cell = xldate_as_tuple(worksheet.cell_value\
(row_index, col_index),workbook.datemode)
                date_cell = date(*date_cell[0:3]).strftime\
('%m/%d/%Y')
                row_list_output.append(date_cell)
                output_worksheet.write(row_index, col_index, date_cell)
            else:
                non_date_cell = worksheet.cell_value\
(row_index,col_index)
                row_list_output.append(non_date_cell)
                output_worksheet.write(row_index, col_index,\
non_date_cell)
output_workbook.save(output_file)

# 处理多个工作簿

pandas 提供了 concat 函数来连接数据框。如果你想把数据框一个一个地垂直堆叠起来，那 么就要设置参数 axis=0。如果你想把数据框一个一个地平行连接起来，那么就要设置参数 axis=1。此外，如果你需要基于某个关键字列连接数据框，pandas 中的 merge 函数可以提 供类似 SQL join 的操作

从上面几段开始就一直只用pandas代码了

import pandas as pd

# 遍历，groupby，try-expect

这里是把遍历获取的dataframe先组合到一起，然后再groupby计算，这样会不会内存里形成一张大表，数据特别大时会不会出错，能支持多大数据量呢

import pandas as pd
import glob
import os,time
from xlrd import open_workbook,xldate_as_tuple

input_path = r"C:\Users\46957\Desktop\pytest\basisofpython\excels"
output_file = r"C:\Users\46957\Desktop\pytest\basisofpython\result\excels\gg3.xlsx"



def walk_dir2(dirname):
    for d in os.listdir(dirname):
        path = os.path.join(dirname, d)
        if os.path.isdir(path):
            for f in walk_dir2(path):
                yield f
        else:
            yield path


filepath = os.path.abspath(input_path)
files = walk_dir2(filepath)

alldataframe = None
data_frames = []

for file in files:
    # 如果是csv格式
    if file.split('.')[-1] == 'csv':
        csv_file = pd.read_csv(file, index_col=None,encoding = 'gb2312')
        data_frames.append(csv_file)
    elif file.split('.')[-1] == 'xlsx' or file.split('.')[-1] == 'xls':
        try:
            all_worksheets = pd.read_excel(file, sheet_name=None, index_col=None)
        except:
            print("---------")
            print(file)
            print("---------")
        else:
            for worksheet_name, data in all_worksheets.items():
                data_frames.append(data)
        finally:
            pass


alldataframe = pd.concat(data_frames, axis=0, ignore_index=True)

# print(alldataframe)
chengji = (alldataframe.groupby('班级').sum()['成绩']).values
banji = alldataframe.groupby('班级').sum().index.tolist()

data = {
    "班级":banji,
    "成绩":chengji
}

# print(data)
fm = pd.DataFrame(data, columns=['班级', '成绩'])


with pd.ExcelWriter(output_file,mode='a+',engine='openpyxl') as writer:
    fm.to_excel(writer, sheet_name='lcs', \
    index=False)
    writer.save()
