# 2.1.1 查找1981年上半年入职的员工有几个人

WHERE DATE_FORMAT(hiredate,"Y")=1981

# 2.2 日期偏移计算【DATE_ADD】

mysql日期计算

mysql里，两个日期不能直接加减、日期也不能与数字加减

# DATE_ADD(日期,INTERVAL 偏移量 时间单位);

# 15天后
SELECT DATE_ADD(NOW(),INTERVAL 15 DAY);

# 300分钟前
SELECT DATE_ADD(NOW(),INTERVAL -300,MINUTE); 

# 可嵌套使用
# 6个月前的3天前,格式yyyy/mm/dd
SELECT 
DATE_FORMAT(
DATE_ADD(DATE_ADD(NOW(),INTERVAL -6 MONTH),INTERVAL -3 DAY)
,"%Y/%m/%d");

# 4 条件函数

# 4.1 IF NULL / IF

IFNULL(表达式,值)

IF(表达式,值1,值2)

例：sales部门发A，其余部门发B，打印所以员工获得的礼品。

SELECT 
	e.ename,e.deptno,d.dname,
	IF(d.dname="SALES","礼品A","礼品B")
FROM t_emp e JOIN t_dept d ON e.deptno=d.deptno;

# 4.2 CASE WHEN THEN ELSE END

if语句只能判断两个值。多了就没用了

CASE
	WHEN 表达式 THEN 值1
	WHEN 表达式 THEN 值2
	...
	ELSE 值N
END

例：每个部门目的地不同。salse部门去P1，ACCOUNTING去P2，RESEARCH部门去P3，查询每名员工的地点。

SELECT 
	e.ename,e.empno,
	CASE
		WHEN d.dname="SALES" THEN "P1"
		WHEN d.dname="ACCOUNTING" THEN "P2"
		WHEN d.dname="RESEARCH" THEN "P3"
	END AS place
FROM t_emp e JOIN t_dept d ON e.deptno=d.deptno;

# 4.3 UPDATE中，更复杂的CASE WHNE THEN

调整：

  1. SALES部门中工龄超过20年，涨幅10%；
  2. SALES部门中工龄不满20年，涨幅5%；
  3. ACCOUNTING部门，+300
  4. RESEARCH部门里低于部门平均底薪，+200
  5. 没有部门的员工，+100
# 左连接，因为没有部门的员工也要保留
UPDATE t_emp e LEFT JOIN t_dept d ON e.deptno=d.deptno   
#不知道4中平均底薪，所以还要连一张表【结果集】
LEFT JOIN 
(SELECT deptno,AVG(sal) AS avg FROM t_emp GROUP BY deptno) t
ON e.deptno=t.deptno

SET e.sal=(
	CASE 
		WHEN d.dname="SALES" AND DATEDIFF(NOW(),e.hiredate)/365>=20 THEN e.sal*1.1
		WHEN d.dname="SALES" AND DATEDIFF(NOW(),e.hiredate)/365<20 THEN e.sal*1.05
		WHEN d.dname="ACCOUNTING" THEN e.sal+300
		WHEN d.dname="RESEARCH" AND e.sal<t.avg THEN e.sal+200
		WHEN e.deptno IS NULL THEN e.sal+100
		ELSE e.sal
	END
);
