# JDBC查询操作 与 SQL注入

# 1.1 JDBC查询操作

新建package包hrapp，在包里新建入口类hunmanResourceApplication

package imooc.jdbc.hrapp;

import imooc.jdbc.hrapp.command.Command;
import imooc.jdbc.hrapp.command.QueryCommad;

import java.util.Scanner;

public class HunmanResourceApplication {
    public static void main(String[] args) {
        System.out.println("1-查询部门员工");
        Scanner in = new Scanner(System.in);
        Integer cmd = in.nextInt();
        /*
        * Integer 与 int 的区别
        * Integer 是 int 的包装类,int 是 java 中的基本数据类型。
        * Integer 对象必须是实例化后才能使用,而 int 变量不需要实例化。
        * Integer 实际是对象的引用,当 new 一个 Integer 时,实际上是生成一个指针指向此对象;而 int 则是直接存储数据值。
        * Integer 的默认值是 null,int 的默认值是 0
        */

        //增删查改都是类似的，可以封装成接口，然后通过不同实现类来处理不同功能
        switch (cmd){
            case 1:
                Command command = new QueryCommad();
                command.execute();
                break;
        }
    }
}

package imooc.jdbc.hrapp.command;

import java.sql.*;
import java.util.Scanner;

public class QueryCommad implements Command{
    @Override
    public void execute(){
        System.out.println("请输入部门名称：");
        Scanner in = new Scanner(System.in);
        String pdname = in.next();
        System.out.println("pdname:"+pdname);
        //jdbc五个步骤
        //1. 加载并注册JDBC驱动
        //2. 创建数据库连接
        //3. 创建statement对象
        //4. 遍历查询
        //5. 关闭连接，释放资源
        //还可以对这几步封装，后面再继续展开

        // Connection Statement ResultSet为什么要放外面。
        // 提升ta们的声明周期。另外，ta们都有自己的close方法，可以再finally中用
        Connection conn = null;
        Statement stmt = null;
        //结果集
        ResultSet rs = null;
        try {
            //1. 加载并注册JDBC驱动
            // Class.forName 用于加载指定的JDBC驱动类；本质是通知JDBC注册这个驱动类
            Class.forName("com.mysql.cj.jdbc.Driver");

            //2. 创建数据库连接
            String url = "jdbc:mysql://localhost:3306/imooc?useSSL=false
			&useUnicode=true&characterEncoding=UTF-8
			&serverTimezone=Asia/Shanghai&allowPublicKeyRetrieval=true";
            String user = "root";
            String password = "lcs123456788";
            conn = DriverManager.getConnection(url, user, password);

            //3. 创建Statement对象
            stmt = conn.createStatement();
            //结果集ResultSet
            rs = stmt.executeQuery("select * from employee where dname='"+ pdname +"'");
            //4. 遍历查询结果
            System.out.println("rs.length:"+rs);
            while (rs.next()) {
				//从结果集取值，根据字段索引，jdbc中字段索引从1开始，而非0
                Integer eno = rs.getInt(1); 
                String ename = rs.getString("ename");  //根据字段名
                Float salary = rs.getFloat("salary");
                String dname = rs.getString("dname");
                System.out.println(dname + "-" + eno + "-" + ename + "-" + salary);
            }
        }catch(ClassNotFoundException e){
            e.printStackTrace();
        }catch(Exception e){
            e.printStackTrace();
        }finally {
            //5. 关闭连接,释放资源
            try {
                if(rs != null){
                    rs.close();
                }
            } catch (SQLException e) {
                e.printStackTrace();
            }

            try {
                if(stmt != null){
                    stmt.close();
                }
            } catch (SQLException e) {
                e.printStackTrace();
            }

            try {
                if(conn != null && !conn.isClosed() ) {
                    conn.close();
                }
            } catch (SQLException e) {
                e.printStackTrace();
            }

            //通常只需要以下操作就可以关闭连接，释放资源.因为物理连接关闭，所有资源就会被释放。
			//但学习中，写完整就是以上，需要判断是否为空，和三个close.
            // conn.close();

        }

# 1.2 SQL注入攻击

  • sql注入攻击是指利用sql漏洞越权获取数据的黑客行为
  • sql注入攻击根源是未对原始sql中的敏感字符做处理

# 1.2.1 演示SQL注入攻击

将输入接收从 in.next() 改为 in.nextLine() 即 接收整行数据





 


...
System.out.println("请输入部门名称：");
Scanner in = new Scanner(System.in);
//String pdname = in.next();
String pdname = in.nextLine();
...

运行时，输入' or 1=1 or 1=',即sql语句为select * from employee where dname='' or 1=1 or 1='' ，如此筛选条件失效。

原因是 ，输入数据没有将特殊字符，如'进行处理，并将其作为sql语句的一部分，拼接运行了。

# 1.2.2 PreparedStatement 【解决sql注入】

放弃Statement改用PrepareStatement预处理SQL。

  • PrepareStatement预编译Statement是Statement的子接口。
  • PrepareStatement对SQL进行参数化，预防SQL注入攻击
  • PrepareStatement比Statement执行效率更高

String sql = "select * from employee where dname=?";中?作为一个特殊字符，如占位，参数化sql。

PrepareStatement pstm = conn.prepareStatement(sql)中， ?要被PrepareStatement解析的。

通过调用setString()方法设置对应参数，参数从1开始。

pstmt.setString(1,pdname);即第一个?的值是pdname

多参数的PrepareStatement

首先要知道，prepareStatement设置参数除了setString方法外还有好多中setInt()、setFloat()、setBoolean()、等等

sql语句添加多个?,String sql = "select * from employee where dname= ? and eno > ?";

设置参数pstmt.setInt(2,3500);

为什么参数化形式 比拼接字符串的形式 要高效呢？

拼接字符串下，每执行一次sql，sql文本都要解析一次rs = stmt.executeQuery("select * from employee where dname='"+ pdname +"'");

而参数化下，不同的?不同带入到prepareStament中，并被prepareStament缓存。即使运行很多次，对sql文本的解析仅一次。

preparestament sql 错误的使用方式

  • sql字段名不能用参数化?替代的，即?只能用在值上。 select * from employee where ?='abc' ❌
  • preparestament 中sql参数不能二次计算 。select * from employee where salary = ?+100 ❌
  • 字段不能参数化 select ? from employee where ename=? 第一个❌ 第二个✔
